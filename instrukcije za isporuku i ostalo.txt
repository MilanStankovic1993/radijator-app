I ubaci sledeći sadržaj:

bash
Copy
Edit
#!/bin/bash

echo "=== DEPLOY STARTED ==="

# Idi u direktorijum projekta
cd /var/www/Projekat-Radijator-Inzenjering || exit

# Povuci najnoviji kod
echo "-> Pulling latest code..."
git pull origin main

# Postavi prava
echo "-> Setting permissions..."
chown -R www-data:www-data storage bootstrap/cache
chmod -R 775 storage bootstrap/cache

# Install dependencies
echo "-> Installing Composer dependencies..."
composer install --no-interaction --prefer-dist --optimize-autoloader

# Clear and cache Laravel stuff
echo "-> Clearing and caching config..."
php artisan config:clear
php artisan route:clear
php artisan view:clear

php artisan config:cache
php artisan route:cache
php artisan view:cache

# Migrate and seed
echo "-> Running migrations and seeders..."
php artisan migrate --force
php artisan db:seed --force

# Restart services (ako koristiš php-fpm i nginx)
echo "-> Restarting PHP-FPM and Nginx..."
systemctl restart php8.3-fpm
systemctl restart nginx

# Zabeleži datum deploy-a
echo "Deployed at $(date)" >> storage/logs/deploy.log

echo "=== DEPLOY COMPLETE ==="
